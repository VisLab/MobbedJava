001    package edu.utsa.mobbed;
002    
003    import java.sql.Connection;
004    import java.util.UUID;
005    
006    /**
007     * Stores metadata information associated with a dataset.
008     * 
009     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
010     * 
011     */
012    public class Metadata {
013    
014            private Attributes atb;
015            private UUID datasetUuid;
016            private Structures modalityStruct;
017            private Connection dbCon;
018            private String metadataField;
019            private Structures metamodalityStruct;
020            private String modalityName;
021    
022            /**
023             * Creates a Metadata object.
024             * 
025             * @param dbCon
026             *            a connection to the database
027             */
028            public Metadata(Connection dbCon) {
029                    this.dbCon = dbCon;
030                    this.datasetUuid = null;
031                    this.metadataField = null;
032                    this.modalityName = null;
033            }
034    
035            /**
036             * Add the attribute to a batch.
037             * 
038             * @param fieldName
039             *            name of the field
040             * @param numericValues
041             *            the numeric values of the attribute
042             * @param values
043             *            the string values of the attribute
044             * @throws MobbedException
045             *             if an error occurs
046             */
047            public void addAttribute(String fieldName, Double[] numericValues,
048                            String[] values) throws MobbedException {
049                    addNewStructure(fieldName);
050                    UUID structureUUID = metamodalityStruct.getChildStructUuid(fieldName);
051                    for (int i = 0; i < numericValues.length; i++) {
052                            atb.reset(UUID.randomUUID(), null, null, datasetUuid, "datasets",
053                                            structureUUID, numericValues[i], values[i]);
054                            atb.addToBatch();
055                    }
056            }
057    
058            /**
059             * Sets the class fields of a Metadata object.
060             * 
061             * @param modalityName
062             *            the name of the modality
063             * @param datasetUuid
064             *            the UUID of the dataset
065             * @param metadataField
066             *            the name of the field that contains the metadata
067             * @throws MobbedException
068             *             if an error occurs
069             */
070            public void reset(String modalityName, String datasetUuid,
071                            String metadataField) throws MobbedException {
072                    this.modalityName = modalityName;
073                    this.datasetUuid = UUID.fromString(datasetUuid);
074                    this.metadataField = metadataField;
075                    atb = new Attributes(dbCon);
076            }
077    
078            /**
079             * Saves all metadata as a batch. All metadata in the batch will be
080             * successfully written or the operation will be aborted.
081             * 
082             * @throws MobbedException
083             *             if an error occurs
084             */
085            public void save() throws MobbedException {
086                    atb.save();
087            }
088    
089            /**
090             * Adds a field to the structures table if it does not already exist.
091             * 
092             * @param fieldName
093             *            the name of the field
094             * @throws MobbedException
095             *             if an error occurs
096             */
097            private void addNewStructure(String fieldName) throws MobbedException {
098                    modalityStruct = Structures.retrieve(dbCon, modalityName,
099                                    UUID.fromString(ManageDB.noParentUuid), true);
100                    metamodalityStruct = Structures.retrieve(dbCon, metadataField,
101                                    modalityStruct.getStructureUuid(), true);
102                    if (!metamodalityStruct.containsChild(fieldName)) {
103                            Structures newChild = new Structures(dbCon);
104                            newChild.reset(UUID.randomUUID(), fieldName,
105                                            metamodalityStruct.getStructureUuid());
106                            newChild.save();
107                            metamodalityStruct.addChild(fieldName, newChild.getStructureUuid());
108                    }
109            }
110    }

077            /**
078             * Inserts or updates rows in the database. To insert rows, do not assign
079             * values to the key columns. To update rows, assign values to the key
080             * columns that already exist in the database.
081             * 
082             * @param tableName
083             *            the name of the database table
084             * @param columnNames
085             *            the names of the columns in the database table
086             * @param columnValues
087             *            the values of the columns that are not double precision in the
088             *            database table
089             * @param doubleColumnNames
090             *            the names of the columns that are double precision in the
091             *            database table
092             * @param doubleValues
093             *            the values of the columns that are double precision in the
094             *            database table
095             * @return the keys of the rows that were inserted or updated
096             * @throws MobbedException
097             *             if an error occurs
098             */
099            public String[] addRows(String tableName, String[] columnNames,
100                            String[][] columnValues, String[] doubleColumnNames,
101                            Double[][] doubleValues) throws MobbedException {
102                    validateTableName(tableName);
103                    validateColumnNames(columnNames);
104                    int numRows = columnValues.length;
105                    int numValues = columnValues[0].length;
106                    int doubleIndex = 0;
107                    Double[] currentDoubleValues;
108                    String[] keyList = new String[numRows];
109                    ArrayList<Integer> keyIndexes = findKeyIndexes(tableName, columnNames);
110                    String insertQry = constructInsertQuery(tableName, columnNames);
111                    String updateQry = constructUpdateQuery(keyIndexes, tableName,
112                                    columnNames);
113                    try {
114                            PreparedStatement insertStmt = connection
115                                            .prepareStatement(insertQry);
116                            PreparedStatement updateStmt = connection
117                                            .prepareStatement(updateQry);
118                            for (int i = 0; i < numRows; i++) {
119                                    for (int j = 0; j < numValues; j++) {
120                                            if (!isEmpty(columnValues[i][j]))
121                                                    validateColumnValue(columnNames[j], columnValues[i][j]);
122                                            else {
123                                                    if (!keyIndexes.contains(j)
124                                                                    && !columnNames[j]
125                                                                                    .equals("dataset_session_uuid"))
126                                                            columnValues[i][j] = getDefaultValue(columnNames[j]);
127                                            }
128                                    }
129                                    if (!isEmpty(doubleValues))
130                                            currentDoubleValues = doubleValues[doubleIndex++];
131                                    else
132                                            currentDoubleValues = null;
133                                    if (keysExist(keyIndexes, tableName, columnNames,
134                                                    columnValues[i])) {
135                                            setUpdateStatementValues(keyIndexes, updateStmt,
136                                                            columnNames, columnValues[i], currentDoubleValues);
137                                            if (verbose)
138                                                    System.out.println(updateStmt);
139                                            updateStmt.addBatch();
140                                    } else {
141                                            columnValues[i] = generateKeys(keyIndexes, tableName,
142                                                            columnNames, columnValues[i]);
143                                            setInsertStatementValues(insertStmt, columnNames,
144                                                            columnValues[i], currentDoubleValues);
145                                            if (verbose)
146                                                    System.out.println(insertStmt);
147                                            insertStmt.addBatch();
148    
149                                    }
150                                    keyList[i] = addKeyValue(keyIndexes, columnValues[i]);
151                            }
152                            insertStmt.executeBatch();
153                            updateStmt.executeBatch();
154                    } catch (SQLException me) {
155                            throw new MobbedException(
156                                            "Could not insert or update row(s) in the database\n"
157                                                            + me.getMessage());
158                    }
159                    return keyList;
160            }

162            /**
163             * Checks the dataset namespace and name combination. If isUnique is set to
164             * true and the combination already exist then an exception will be thrown.
165             * If isUnique is false and the combination already exist then the dataset
166             * version will be incremented signifying a duplicate dataset.
167             * 
168             * @param isUnique
169             *            if the dataset namespace and name combination is unique
170             *            combination
171             * @param datasetName
172             *            the name of the dataset
173             * @param datasetNamespace
174             *            the namespace of the dataset
175             * @return the version number of the dataset
176             * @throws MobbedException
177             *             if an error occurs
178             */
179            public int checkDatasetVersion(boolean isUnique, String datasetNamespace,
180                            String datasetName) throws MobbedException {
181                    String query = "SELECT MAX(DATASET_VERSION) AS LATESTVERSION"
182                                    + " FROM DATASETS WHERE DATASET_NAMESPACE = ? AND DATASET_NAME = ?";
183                    int version;
184                    try {
185                            PreparedStatement selStmt = connection.prepareStatement(query);
186                            selStmt.setString(1, datasetNamespace);
187                            selStmt.setString(2, datasetName);
188                            ResultSet rs = selStmt.executeQuery();
189                            rs.next();
190                            version = rs.getInt(1);
191                    } catch (SQLException ex) {
192                            throw new MobbedException(
193                                            "Could not execute query to determine dataset version\n"
194                                                            + ex.getMessage());
195                    }
196                    if (isUnique && version > 0)
197                            throw new MobbedException("dataset version is not unique");
198                    return version + 1;
199            }

598            /**
599             * Constructs a query based on search criteria.
600             * 
601             * @param tableName
602             *            the name of the database table
603             * @param regExp
604             *            on if regular expressions are allowed, off if otherwise
605             * @param tags
606             *            the tag search criteria
607             * @param attributes
608             *            the attribute search criteria
609             * @param columnNames
610             *            the names of the columns
611             * @param columnValues
612             *            the values of the columns
613             * @return a string query
614             * @throws MobbedException
615             *             if an error occurs
616             */
617            private String constructQualificationQuery(String tableName, String regExp,
618                            String[][] tags, String[][] attributes, String[] columnNames,
619                            String[][] columnValues) throws MobbedException {
620                    String qry = "";
621                    String closer = "";
622                    String[] keys = keyMap.get(tableName);
623                    if (tags != null || attributes != null) {
624                            qry = " WHERE " + keys[0] + " IN (";
625                            closer = ")";
626                    }
627                    if (tags != null)
628                            qry += constructTagAttributesQuery(regExp, "Tags", tags);
629                    if (attributes != null) {
630                            if (tags != null)
631                                    qry += " INTERSECT ";
632                            qry += constructTagAttributesQuery(regExp, "Attributes", attributes);
633                    }
634                    if (columnNames != null) {
635                            if (tags != null || attributes != null)
636                                    qry += " INTERSECT SELECT " + keys[0] + " FROM " + tableName
637                                                    + " WHERE ";
638                            else
639                                    qry += " WHERE ";
640                            qry += constructTableQuery(regExp, tableName, columnNames,
641                                            columnValues);
642                    }
643                    qry += closer;
644                    return qry;
645            }

669            /**
670             * Constructs a query associated with a table in the database.
671             * 
672             * @param regExp
673             *            on if regular expressions are allowed, off if otherwise
674             * @param tableName
675             *            the name of the database table
676             * @param columnNames
677             *            the names of the columns
678             * @param columnValues
679             *            the values of the columns
680             * @return a query string
681             */
682            private String constructTableQuery(String regExp, String tableName,
683                            String[] columnNames, String[][] columnValues) {
684                    String type;
685                    String columnName;
686                    String qry = "";
687                    int numColumns = columnNames.length;
688                    for (int i = 0; i < numColumns; i++) {
689                            type = typeMap.get(columnNames[i]);
690                            // Case insensitive fix
691                            if (type.equalsIgnoreCase("character varying"))
692                                    columnName = " UPPER(" + columnNames[i] + ")";
693                            else
694                                    columnName = columnNames[i];
695                            int numValues = columnValues[i].length;
696                            if (type.equalsIgnoreCase("character varying")
697                                            && regExp.equalsIgnoreCase("on")) {

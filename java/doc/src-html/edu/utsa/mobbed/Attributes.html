001    package edu.utsa.mobbed;
002    
003    import java.util.UUID;
004    import java.sql.*;
005    
006    /**
007     * Handler class for ATTRIBUTES table. Attributes, which do not have timestamps,
008     * may encapsulate entity metadata or other characteristics as well as time
009     * independent data.
010     * 
011     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
012     * 
013     */
014    public class Attributes {
015    
016            private UUID attributeUuid;
017            private UUID entityUuid;
018            private String entityClass;
019            private UUID organizationalUuid;
020            private String organizationalClass;
021            private UUID structureUuid;
022            private Double numericValue;
023            private String value;
024            private PreparedStatement insertStmt;
025            public static String insertQry = "INSERT INTO ATTRIBUTES "
026                            + "(ATTRIBUTE_UUID, ATTRIBUTE_ENTITY_UUID, ATTRIBUTE_ENTITY_CLASS, ATTRIBUTE_ORGANIZATIONAL_UUID, "
027                            + "ATTRIBUTE_ORGANIZATIONAL_CLASS, ATTRIBUTE_STRUCTURE_UUID,  ATTRIBUTE_NUMERIC_VALUE, ATTRIBUTE_VALUE)"
028                            + " VALUES (?, ?, ?, ?, ?, ?, ?, ?)";
029    
030            /**
031             * Creates a Attributes object.
032             * 
033             * @param dbCon
034             *            a connection to the database
035             * @throws MobbedException
036             *             if an error occurs
037             */
038            public Attributes(Connection dbCon) throws MobbedException {
039                    attributeUuid = null;
040                    entityUuid = null;
041                    entityClass = null;
042                    organizationalUuid = null;
043                    organizationalClass = null;
044                    structureUuid = null;
045                    numericValue = null;
046                    value = null;
047                    try {
048                            insertStmt = dbCon.prepareStatement(insertQry);
049                    } catch (SQLException ex) {
050                            throw new MobbedException(
051                                            "Could not create prepared statement for Attributes object\n"
052                                                            + ex.getMessage());
053                    }
054            }
055    
056            /**
057             * Adds a attribute to a batch.
058             * 
059             * @throws MobbedException
060             *             if an error occurs
061             */
062            public void addToBatch() throws MobbedException {
063                    try {
064                            insertStmt.setObject(1, attributeUuid, Types.OTHER);
065                            insertStmt.setObject(2, entityUuid, Types.OTHER);
066                            insertStmt.setString(3, entityClass);
067                            insertStmt.setObject(4, organizationalUuid, Types.OTHER);
068                            insertStmt.setString(5, organizationalClass);
069                            insertStmt.setObject(6, structureUuid, Types.OTHER);
070                            insertStmt.setObject(7, numericValue);
071                            insertStmt.setString(8, value);
072                            insertStmt.addBatch();
073                    } catch (SQLException ex) {
074                            throw new MobbedException("Could not add attribute to batch\n"
075                                            + ex.getMessage());
076                    }
077            }
078    
079            /**
080             * Sets the class fields of a Attributes object.
081             * 
082             * @param attributeUuid
083             *            UUID of the attribute
084             * @param entityUuid
085             *            UUID of the entity associated with the attribute
086             * @param entityClass
087             *            name of the entity table associated with the attribute
088             * @param organizationalUuid
089             *            UUID of the organization associated with the attribute
090             * @param organizationalClass
091             *            name of the organizational table associated with the attribute
092             * @param structureUuid
093             *            UUID of the structure associated with the attribute
094             * @param numericValue
095             *            numeric representation of the attribute value
096             * @param value
097             *            string representation of the attribute value
098             */
099            public void reset(UUID attributeUuid, UUID entityUuid, String entityClass,
100                            UUID organizationalUuid, String organizationalClass,
101                            UUID structureUuid, Double numericValue, String value) {
102                    this.attributeUuid = attributeUuid;
103                    this.entityUuid = entityUuid;
104                    this.entityClass = entityClass;
105                    this.organizationalUuid = organizationalUuid;
106                    this.organizationalClass = organizationalClass;
107                    this.structureUuid = structureUuid;
108                    this.numericValue = numericValue;
109                    this.value = value;
110            }
111    
112            /**
113             * Saves all attributes as a batch. All attributes in the batch will be
114             * successfully written or the operation will be aborted.
115             * 
116             * @throws MobbedException
117             *             if an error occurs
118             */
119            public void save() throws MobbedException {
120                    try {
121                            insertStmt.executeBatch();
122                    } catch (SQLException ex) {
123                            throw new MobbedException("Could not save the attributes\n"
124                                            + ex.getMessage());
125                    }
126            }
127    }

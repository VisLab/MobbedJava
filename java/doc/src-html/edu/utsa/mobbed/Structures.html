001    package edu.utsa.mobbed;
002    
003    import java.util.HashMap;
004    import java.util.UUID;
005    import java.sql.*;
006    
007    /**
008     * Handler class for STRUCTURES table. The STRUCTURES table encapsulates the
009     * organization of the dataset for insertion into or extraction from a
010     * particular access layer. Structures form a forest for categorization and
011     * reconstruction.
012     * 
013     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
014     * 
015     */
016    
017    public class Structures {
018    
019            private Connection dbCon;
020            private HashMap<String, UUID> children;
021            private UUID parentUuid;
022            private String structureName;
023            private UUID structureUuid;
024            private String structurePath;
025            private static final String insertQry = "INSERT INTO STRUCTURES "
026                            + " (STRUCTURE_UUID, STRUCTURE_NAME, STRUCTURE_PARENT_UUID, STRUCTURE_PATH)"
027                            + " VALUES (?, ?, ?, ?)";
028            private static final String selectQry = "SELECT STRUCTURE_NAME, STRUCTURE_PARENT_UUID FROM STRUCTURES "
029                            + "WHERE STRUCTURE_UUID = ?";
030    
031            /**
032             * Creates a Structures object.
033             * 
034             * @param dbCon
035             *            a connection to the database
036             */
037            public Structures(Connection dbCon) {
038                    this.dbCon = dbCon;
039                    structureUuid = null;
040                    structureName = null;
041                    parentUuid = null;
042                    structurePath = null;
043                    children = new HashMap<String, UUID>();
044            }
045    
046            /**
047             * Adds a child structure to the hashmap.
048             * 
049             * @param childName
050             *            the name of the child structure
051             * @param childUuid
052             *            the UUID of the child structure
053             */
054            public void addChild(String childName, UUID childUuid) {
055                    children.put(childName, childUuid);
056            }
057    
058            /**
059             * Checks if the hashmap contains the child structure.
060             * 
061             * @param childName
062             *            the name of the child structure
063             * @return true if the child structure is in the hashmap, false if otherwise
064             */
065            public boolean containsChild(String childName) {
066                    return children.containsKey(childName);
067            }
068    
069            /**
070             * Gets the UUID of the child structure.
071             * 
072             * @param childName
073             *            the name of the child structure
074             * @return the UUID of the child structure
075             */
076            public UUID getChildStructUuid(String childName) {
077                    return children.get(childName);
078            }
079    
080            /**
081             * Gets the UUID of the structure.
082             * 
083             * @return the UUID of the structure
084             */
085            public UUID getStructureUuid() {
086                    return structureUuid;
087            }
088    
089            /**
090             * Sets the class fields of a Structures object.
091             * 
092             * @param structureUuid
093             *            the UUID of the structure
094             * @param structureName
095             *            the name of the structure
096             * @param parentUuid
097             *            UUID of the parent structure
098             * @throws MobbedException
099             *             if an error occurs
100             */
101            public void reset(UUID structureUuid, String structureName, UUID parentUuid)
102                            throws MobbedException {
103                    this.structureUuid = structureUuid;
104                    this.structureName = structureName;
105                    this.parentUuid = parentUuid;
106                    structurePath = generateStructurePath();
107            }
108    
109            /**
110             * Saves a structure to the database.
111             * 
112             * @throws MobbedException
113             *             if an error occurs
114             */
115            public void save() throws MobbedException {
116                    try {
117                            PreparedStatement insertStmt = dbCon.prepareStatement(insertQry);
118                            insertStmt.setObject(1, structureUuid, Types.OTHER);
119                            insertStmt.setString(2, structureName);
120                            insertStmt.setObject(3, parentUuid, Types.OTHER);
121                            insertStmt.setString(4, structurePath);
122                            insertStmt.execute();
123                    } catch (SQLException ex) {
124                            throw new MobbedException("Could not save the structure\n"
125                                            + ex.getMessage());
126                    }
127            }
128    
129            /**
130             * Generates a structure path based on it's hierarchy.
131             * 
132             * @return the path of the structure
133             * @throws MobbedException
134             *             if an error occurs
135             */
136            private String generateStructurePath() throws MobbedException {
137                    String structurePath = "/" + structureName;
138                    UUID currentParentUuid = parentUuid;
139                    try {
140                            PreparedStatement pstmt = dbCon.prepareStatement(selectQry);
141                            ResultSet rs = null;
142                            while (!UUID.fromString(ManageDB.noParentUuid).equals(
143                                            currentParentUuid)) {
144                                    pstmt.setObject(1, currentParentUuid);
145                                    rs = pstmt.executeQuery();
146                                    if (rs.next()) {
147                                            structurePath = "/" + rs.getString(1) + structurePath;
148                                            currentParentUuid = UUID.fromString(rs.getString(2));
149                                    }
150                            }
151                    } catch (SQLException ex) {
152                            throw new MobbedException("Could not generate the structure path\n"
153                                            + ex.getMessage());
154                    }
155                    return structurePath;
156            }
157    
158            /**
159             * Retrieves the children of the structure from the database.
160             * 
161             * @param dbCon
162             *            a connection to a Mobbed database
163             * @throws MobbedException
164             *             if an error occurs
165             */
166            private void retrieveChildren(Connection dbCon) throws MobbedException {
167                    String query = "SELECT STRUCTURE_NAME, STRUCTURE_UUID FROM STRUCTURES WHERE STRUCTURE_PARENT_UUID = ?";
168                    try {
169                            PreparedStatement pstmt = dbCon.prepareStatement(query);
170                            pstmt.setObject(1, structureUuid, Types.OTHER);
171                            ResultSet rs = pstmt.executeQuery();
172                            while (rs.next())
173                                    children.put(rs.getString(1), UUID.fromString(rs.getString(2)));
174                    } catch (SQLException ex) {
175                            throw new MobbedException(
176                                            "Could not retrieve the structure's children\n"
177                                                            + ex.getMessage());
178                    }
179            }
180    
181            /**
182             * Retrieves a structure with its children.
183             * 
184             * @param dbCon
185             *            a connection to a Mobbed database
186             * @param structureName
187             *            the name of the structure
188             * @param parentUuid
189             *            the UUID of the parent structure
190             * @param retrieveChildren
191             *            retrieves the children of the structure if true
192             * @throws MobbedException
193             *             if an error occurs
194             */
195            public static Structures retrieve(Connection dbCon, String structureName,
196                            UUID parentUuid, boolean retrieveChildren) throws MobbedException {
197                    String query = "SELECT STRUCTURE_UUID FROM STRUCTURES"
198                                    + " WHERE STRUCTURE_NAME = ? AND STRUCTURE_PARENT_UUID = ?";
199                    Structures s = new Structures(dbCon);
200                    try {
201                            PreparedStatement pstmt = dbCon.prepareStatement(query);
202                            pstmt.setString(1, structureName);
203                            pstmt.setObject(2, parentUuid, Types.OTHER);
204                            ResultSet rs = pstmt.executeQuery();
205                            if (rs.next())
206                                    s.reset(UUID.fromString(rs.getString(1)), structureName,
207                                                    parentUuid);
208                            else {
209                                    s.reset(UUID.randomUUID(), structureName, parentUuid);
210                                    s.save();
211                            }
212                            if (retrieveChildren)
213                                    s.retrieveChildren(dbCon);
214                    } catch (SQLException ex) {
215                            throw new MobbedException("Could not retrieve the structure\n"
216                                            + ex.getMessage());
217                    }
218                    return s;
219            }
220    }

001    package edu.utsa.mobbed;
002    
003    import java.sql.*;
004    import java.util.UUID;
005    
006    /**
007     * Handler class for EVENTS table. Event entities have a name, a starting time,
008     * and an ending time. Each event is associated with a particular target entity.
009     * 
010     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
011     * 
012     */
013    
014    public class Events {
015    
016            private Attributes atb;
017            private double[] certainties;
018            private UUID datasetUuid;
019            private Structures modalityStruct;
020            private Connection dbCon;
021            private String eventField;
022            private Structures eventStruct;
023            private UUID[] eventUuids;
024            private EventTypes evType;
025            private String[] existingUuids;
026            private PreparedStatement insertStmt;
027            private String modalityName;
028            private long[] positions;
029            private double[] startTimes;
030            private double[] endTimes;
031            private String[] types;
032            private String[] uniqueTypes;
033            private String[] parentUuids;
034            private static final String insertQry = "INSERT INTO EVENTS (EVENT_UUID, EVENT_ENTITY_UUID, "
035                            + "EVENT_ENTITY_CLASS, EVENT_TYPE_UUID, EVENT_PARENT_UUID, EVENT_POSITION, EVENT_START_TIME,"
036                            + " EVENT_END_TIME, EVENT_CERTAINTY) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)";
037    
038            /**
039             * Creates a Events object.
040             * 
041             * @param dbCon
042             *            a connection to the database
043             */
044            public Events(Connection dbCon) {
045                    this.dbCon = dbCon;
046                    this.datasetUuid = null;
047                    this.modalityName = null;
048                    this.types = null;
049                    this.positions = null;
050                    this.startTimes = null;
051                    this.endTimes = null;
052                    this.certainties = null;
053                    this.uniqueTypes = null;
054                    this.modalityStruct = null;
055            }
056    
057            /**
058             * Add the attribute to a batch.
059             * 
060             * @param fieldName
061             *            the field name of the attribute
062             * @param numericValues
063             *            the numeric values of the attribute
064             * @param values
065             *            the string values of the attribute
066             * @throws MobbedException
067             *             if an error occurs
068             */
069            public void addAttribute(String fieldName, Double[] numericValues,
070                            String[] values) throws MobbedException {
071                    addNewStructure(fieldName);
072                    UUID structureUUID = eventStruct.getChildStructUuid(fieldName);
073                    for (int i = 0; i < types.length; i++) {
074                            atb.reset(UUID.randomUUID(), eventUuids[i], "events", datasetUuid,
075                                            "datasets", structureUUID, numericValues[i], values[i]);
076                            atb.addToBatch();
077                    }
078            }
079    
080            /**
081             * Adds the events to a batch. The original events should be stored prior to
082             * storing the events that are derived from them.
083             * 
084             * @return the UUIDs of the events that were stored
085             * @throws MobbedException
086             *             if an error occurs
087             */
088            public String[] addEvents() throws MobbedException {
089                    addNewTypes();
090                    String entityClass = "datasets";
091                    eventUuids = new UUID[types.length];
092                    String[] stringEventUuids = new String[types.length];
093                    try {
094                            insertStmt = dbCon.prepareStatement(insertQry);
095                            for (int i = 0; i < types.length; i++) {
096                                    eventUuids[i] = UUID.randomUUID();
097                                    stringEventUuids[i] = eventUuids[i].toString();
098                                    insertStmt.setObject(1, eventUuids[i], Types.OTHER);
099                                    insertStmt.setObject(2, datasetUuid, Types.OTHER);
100                                    insertStmt.setString(3, entityClass);
101                                    insertStmt.setObject(4,
102                                                    evType.getEventTypeUuid(types[i].toUpperCase()),
103                                                    Types.OTHER);
104                                    insertStmt.setObject(5, parentUuids[i], Types.OTHER);
105                                    insertStmt.setLong(6, positions[i]);
106                                    insertStmt.setDouble(7, startTimes[i]);
107                                    insertStmt.setDouble(8, endTimes[i]);
108                                    insertStmt.setDouble(9, certainties[i]);
109                                    insertStmt.addBatch();
110                            }
111                    } catch (SQLException ex) {
112                            throw new MobbedException("Could not add the event to the batch\n"
113                                            + ex.getMessage());
114                    }
115                    return stringEventUuids;
116            }
117    
118            /**
119             * Adds a new event type if it does not already exist. Event types should be
120             * reused when storing datasets that have event types with the same meaning.
121             * 
122             * @return the UUIDs of the event types
123             * @throws MobbedException
124             *             if an error occurs
125             */
126            public String[] addNewTypes() throws MobbedException {
127                    if (existingUuids != null)
128                            evType.retrieveMap(existingUuids);
129                    EventTypes newEventType = new EventTypes(dbCon);
130                    for (int i = 0; i < uniqueTypes.length; i++) {
131                            if (!evType.containsEventType(uniqueTypes[i].toUpperCase())) {
132                                    newEventType.reset(uniqueTypes[i], null);
133                                    newEventType.save();
134                                    evType.addEventType(uniqueTypes[i].toUpperCase(),
135                                                    newEventType.getEventTypeUuid());
136                            }
137                    }
138                    return evType.getStringValues();
139            }
140    
141            /**
142             * Sets class fields of a Events object.
143             * 
144             * @param modalityName
145             *            the modality name associated with the events
146             * @param datasetUuid
147             *            the UUID of the dataset associated with the events
148             * @param eventField
149             *            the name of the field that contains the events
150             * @param uniqueTypes
151             *            the unique event types
152             * @param types
153             *            all of the event types
154             * @param positions
155             *            the positions of the events
156             * @param startTimes
157             *            the start times of the events
158             * @param endTimes
159             *            the end times of the events
160             * @param certainties
161             *            the certainties of the events
162             * @param existingUuids
163             *            the UUIDs of the event types that will be reused
164             * @param parentUuids
165             *            the UUIDs of the original events in which these events were
166             *            derived from
167             * @throws MobbedException
168             *             if an error occurs
169             */
170            public void reset(String modalityName, String datasetUuid,
171                            String eventField, String uniqueTypes[], String[] types,
172                            long[] positions, double[] startTimes, double[] endTimes,
173                            double[] certainties, String[] existingUuids, String[] parentUuids)
174                            throws MobbedException {
175                    this.modalityName = modalityName;
176                    this.datasetUuid = UUID.fromString(datasetUuid);
177                    this.eventField = eventField;
178                    this.types = types;
179                    this.positions = positions;
180                    this.startTimes = startTimes;
181                    this.endTimes = endTimes;
182                    this.certainties = certainties;
183                    this.uniqueTypes = uniqueTypes;
184                    this.existingUuids = existingUuids;
185                    this.parentUuids = parentUuids;
186                    atb = new Attributes(dbCon);
187                    evType = new EventTypes(dbCon);
188            }
189    
190            /**
191             * Saves all events as a batch. All events in the batch will be successfully
192             * written or the operation will be aborted.
193             * 
194             * @throws MobbedException
195             *             if an error occurs
196             */
197            public void save() throws MobbedException {
198                    atb.save();
199                    try {
200                            insertStmt.executeBatch();
201                    } catch (SQLException ex) {
202                            throw new MobbedException("Could not save the events\n"
203                                            + ex.getMessage());
204                    }
205            }
206    
207            /**
208             * Adds a field to the structures table if it does not already exist.
209             * 
210             * @param fieldName
211             *            the name of the field
212             * @throws MobbedException
213             *             if an error occurs
214             */
215            private void addNewStructure(String fieldName) throws MobbedException {
216                    modalityStruct = Structures.retrieve(dbCon, modalityName,
217                                    UUID.fromString(ManageDB.noParentUuid), false);
218                    eventStruct = Structures.retrieve(dbCon, eventField,
219                                    modalityStruct.getStructureUuid(), true);
220                    if (!eventStruct.containsChild(fieldName)) {
221                            Structures newChild = new Structures(dbCon);
222                            newChild.reset(UUID.randomUUID(), fieldName,
223                                            eventStruct.getStructureUuid());
224                            newChild.save();
225                            eventStruct.addChild(fieldName, newChild.getStructureUuid());
226                    }
227            }
228    
229    }

001    package edu.utsa.mobbed;
002    
003    import java.util.UUID;
004    import java.sql.*;
005    
006    /**
007     * Handler class for ELEMENTS table. Element entities identify time stamped data
008     * streams. An element may correspond to a single entity or may contain sub
009     * elements.
010     * 
011     * 
012     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
013     * 
014     */
015    
016    public class Elements {
017            private Attributes atb;
018            private UUID datasetUuid;
019            private Structures modalityStruct;
020            private Connection dbCon;
021            private String[] elementDescriptions;
022            private String elementField;
023            private String[] elementLabels;
024            private long[] elementPositions;
025            private Structures elementStruct;
026            private UUID[] elementUuids;
027            private String groupLabel;
028            private UUID groupUuid;
029            private PreparedStatement insertStmt;
030            private String modalityName;
031            private static final String insertElementQry = "INSERT INTO ELEMENTS "
032                            + "(ELEMENT_UUID, ELEMENT_LABEL, ELEMENT_ORGANIZATIONAL_UUID, ELEMENT_ORGANIZATIONAL_CLASS, ELEMENT_PARENT_UUID, ELEMENT_POSITION, ELEMENT_DESCRIPTION) "
033                            + "VALUES (?, ?, ?, ?, ?, ?, ?)";
034    
035            /**
036             * Creates a Elements object.
037             * 
038             * @param dbCon
039             *            a connection to the database
040             */
041            public Elements(Connection dbCon) {
042                    this.dbCon = dbCon;
043                    this.datasetUuid = null;
044                    this.modalityName = null;
045                    this.groupLabel = null;
046                    this.groupUuid = null;
047                    this.elementField = null;
048                    this.elementLabels = null;
049                    this.elementDescriptions = null;
050                    this.elementPositions = null;
051                    this.modalityStruct = null;
052            }
053    
054            /**
055             * Add the attribute to a batch.
056             * 
057             * @param fieldName
058             *            the field name of the attribute
059             * @param numericValues
060             *            the numeric values of the attribute
061             * @param values
062             *            the string values of the attribute
063             * @throws MobbedException
064             *             if an error occurs
065             */
066            public void addAttribute(String fieldName, Double[] numericValues,
067                            String[] values) throws MobbedException {
068                    addNewStructure(fieldName);
069                    UUID structureUUID = elementStruct.getChildStructUuid(fieldName);
070                    for (int i = 0; i < elementLabels.length; i++) {
071                            atb.reset(UUID.randomUUID(), elementUuids[i], "elements",
072                                            datasetUuid, "datasets", structureUUID, numericValues[i],
073                                            values[i]);
074                            atb.addToBatch();
075                    }
076            }
077    
078            /**
079             * Adds the elements to a batch. A group element will be added to the batch
080             * prior to the child elements if a group label is given.
081             * 
082             * @return the UUIDs of the elements that were added to the batch
083             * @throws MobbedException
084             */
085            public String[] addElements() throws MobbedException {
086                    String organizationalClass = "datasets";
087                    elementUuids = new UUID[elementLabels.length];
088                    String[] stringElementUuids = new String[elementLabels.length];
089                    try {
090                            insertStmt = dbCon.prepareStatement(insertElementQry);
091                            if (groupLabel != null) {
092                                    groupUuid = UUID.randomUUID();
093                                    insertStmt.setObject(1, groupUuid, Types.OTHER);
094                                    insertStmt.setObject(2, groupLabel);
095                                    insertStmt.setObject(3, datasetUuid);
096                                    insertStmt.setString(4, organizationalClass);
097                                    insertStmt.setObject(5, ManageDB.noParentUuid, Types.OTHER);
098                                    insertStmt.setInt(6, 1);
099                                    insertStmt.setObject(7, groupLabel);
100                                    insertStmt.addBatch();
101                            }
102                            for (int i = 0; i < elementLabels.length; i++) {
103                                    elementUuids[i] = UUID.randomUUID();
104                                    stringElementUuids[i] = elementUuids[i].toString();
105                                    insertStmt.setObject(1, elementUuids[i], Types.OTHER);
106                                    insertStmt.setObject(2, elementLabels[i]);
107                                    insertStmt.setObject(3, datasetUuid);
108                                    insertStmt.setString(4, organizationalClass);
109                                    insertStmt.setObject(5, groupUuid, Types.OTHER);
110                                    insertStmt.setLong(6, elementPositions[i]);
111                                    insertStmt.setObject(7, elementDescriptions[i]);
112                                    insertStmt.addBatch();
113                            }
114                    } catch (SQLException ex) {
115                            throw new MobbedException("Could not add elements to the batch\n"
116                                            + ex.getMessage());
117                    }
118                    return stringElementUuids;
119            }
120    
121            /**
122             * Sets the class fields of a Elements object.
123             * 
124             * @param modalityName
125             *            the name of the modality associated with the elements
126             * @param datasetUuid
127             *            the UUID of the dataset associated with the elements
128             * @param elementField
129             *            the field that contains the elements
130             * @param groupLabel
131             *            the label of the parent element
132             * @param elementLabels
133             *            the labels of the elements
134             * @param elementDescriptions
135             *            the descriptions of the elements
136             * @param elementPositions
137             *            the positions of the elements. starting at 1, 2, ...
138             * @throws MobbedException
139             *             if an error occurs
140             */
141            public void reset(String modalityName, String datasetUuid,
142                            String elementField, String groupLabel, String[] elementLabels,
143                            String[] elementDescriptions, long[] elementPositions)
144                            throws MobbedException {
145                    this.modalityName = modalityName;
146                    this.datasetUuid = UUID.fromString(datasetUuid);
147                    this.groupLabel = groupLabel;
148                    this.elementField = elementField;
149                    this.elementLabels = elementLabels;
150                    this.elementDescriptions = elementDescriptions;
151                    this.elementPositions = elementPositions;
152                    atb = new Attributes(dbCon);
153            }
154    
155            /**
156             * Saves all elements as a batch. All elements in the batch will be
157             * successfully written or the operation will be aborted.
158             * 
159             * @throws MobbedException
160             *             if an error occurs
161             */
162            public void save() throws MobbedException {
163                    atb.save();
164                    try {
165                            insertStmt.executeBatch();
166                    } catch (SQLException ex) {
167                            throw new MobbedException("Could not save elements\n"
168                                            + ex.getMessage());
169                    }
170            }
171    
172            /**
173             * Adds a field to the structures table if it does not already exist.
174             * 
175             * @param fieldName
176             *            the name of the field
177             * @throws MobbedException
178             *             if an error occurs
179             */
180            private void addNewStructure(String fieldName) throws MobbedException {
181                    modalityStruct = Structures.retrieve(dbCon, modalityName,
182                                    UUID.fromString(ManageDB.noParentUuid), false);
183                    elementStruct = Structures.retrieve(dbCon, elementField,
184                                    modalityStruct.getStructureUuid(), true);
185                    if (!elementStruct.containsChild(fieldName)) {
186                            Structures newChild = new Structures(dbCon);
187                            newChild.reset(UUID.randomUUID(), fieldName,
188                                            elementStruct.getStructureUuid());
189                            newChild.save();
190                            elementStruct.addChild(fieldName, newChild.getStructureUuid());
191                    }
192            }
193    
194    }

029            /**
030             * Retrieves a blob from the database.
031             * 
032             * @param dbCon
033             *            a connection to the database
034             * @param filename
035             *            the name of the file that the blob will be written to
036             * @param entityUuid
037             *            the entity UUID that is associated
038             * @param additional
039             *            true if the blob is addition data, false if otherwise
040             * @throws MobbedException
041             *             if an error occurs
042             */
043            public static void retrieveBlob(Connection dbCon, String filename,
044                            String entityUuid, boolean additional) throws MobbedException {
045                    String query;
046                    if (additional)
047                            query = "SELECT DATASET_OID FROM DATASETS WHERE DATASET_UUID = ? AND DATASET_OID IS NOT NULL LIMIT 1";
048                    else
049                            query = "SELECT DATADEF_OID FROM DATADEFS WHERE DATADEF_UUID = ? AND DATADEF_OID IS NOT NULL LIMIT 1";
050                    try {
051                            LargeObjectManager lobj = ((org.postgresql.PGConnection) dbCon)
052                                            .getLargeObjectAPI();
053                            PreparedStatement ps = dbCon.prepareStatement(query);
054                            ps.setObject(1, entityUuid, Types.OTHER);
055                            ResultSet rs = ps.executeQuery();
056                            if (rs.next()) {
057                                    long oid = rs.getLong(1);
058                                    LargeObject obj = lobj.open(oid, LargeObjectManager.READ);
059                                    File file = new File(filename);
060                                    FileOutputStream fos = new FileOutputStream(file);
061                                    int fileLength = obj.size();
062                                    byte buf[] = new byte[65536];
063                                    int s, tl = 0;
064                                    int readSize = Math.min(65536, fileLength - tl);
065                                    while ((s = obj.read(buf, 0, readSize)) > 0) {
066                                            fos.write(buf, 0, s);
067                                            tl += s;
068                                            readSize = Math.min(65536, fileLength - tl);
069                                    }
070                                    fos.close();
071                                    obj.close();
072                            }
073                            rs.close();
074                            ps.close();
075                    } catch (Exception ex) {
076                            throw new MobbedException("Could not retrieve the blob\n"
077                                            + ex.getMessage());
078                    }
079            }

081            /**
082             * Retrieves a numeric value data definition.
083             * 
084             * @param dbCon
085             *            a connection to the database
086             * @param datadefUuid
087             *            the data definition UUID that is associated
088             * @return a array that contains the numeric value data
089             * @throws MobbedException
090             *             if an error occurs
091             */
092            public static Object retrieveNumericValue(Connection dbCon,
093                            String datadefUuid) throws MobbedException {
094                    Object numericData = null;
095                    String selectQry = "SELECT NUMERIC_VALUE_DATA_VALUE FROM NUMERIC_VALUES WHERE NUMERIC_VALUE_DEF_UUID = ?";
096                    try {
097                            PreparedStatement insertStmt = dbCon.prepareStatement(selectQry);
098                            insertStmt.setObject(1, UUID.fromString(datadefUuid));
099                            ResultSet rs = insertStmt.executeQuery();
100                            if (rs.next())
101                                    numericData = rs.getArray(1).getArray();
102                    } catch (SQLException ex) {
103                            throw new MobbedException("Could not retrieve the numeric values\n"
104                                            + ex.getMessage());
105                    }
106                    return numericData;
107            }

137            /**
138             * Stores a blob in the database.
139             * 
140             * @param dbCon
141             *            a connection to the database
142             * @param filename
143             *            the name of the file that contains the blob
144             * @param entityUuid
145             *            the entity UUID that is associated
146             * @param additional
147             *            true if the blob is addition data, false if otherwise
148             * @return a oid associated with the blob stored
149             * @throws MobbedException
150             *             if an error occurs
151             */
152            public static long storeBlob(Connection dbCon, String filename,
153                            String entityUuid, boolean additional) throws MobbedException {
154                    long oid = 0;
155                    try {
156                            LargeObjectManager lobj = ((org.postgresql.PGConnection) dbCon)
157                                            .getLargeObjectAPI();
158                            oid = lobj.createLO();
159                            LargeObject obj = lobj.open(oid, LargeObjectManager.WRITE);
160                            File file = new File(filename);
161                            FileInputStream fis = new FileInputStream(file);
162                            long fileLength = file.length();
163                            byte[] buff = new byte[65536];
164                            int s, tl = 0;
165                            int readSize = (int) Math.min(65536, fileLength - tl);
166                            while ((s = fis.read(buff, 0, readSize)) > 0) {
167                                    obj.write(buff, 0, s);
168                                    tl += s;
169                                    readSize = (int) Math.min(65536, fileLength - tl);
170                            }
171                            obj.close();
172                            fis.close();
173                            if (additional)
174                                    createDatasetOid(dbCon, entityUuid, oid);
175                            else
176                                    createdatadefOid(dbCon, entityUuid, oid);
177                    } catch (Exception ex) {
178                            throw new MobbedException("Could not save the blob\n"
179                                            + ex.getMessage());
180                    }
181                    return oid;
182            }

291            /**
292             * Validates a xml value data definition.
293             * 
294             * @param xml
295             *            the xml value that will be validated
296             * @throws MobbedException
297             *             if an error occurs
298             */
299            private static void validateXMLValue(String xml) throws MobbedException {
300                    try {
301                            DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
302                            DocumentBuilder db = dbf.newDocumentBuilder();
303                            InputSource is = new InputSource();
304                            Pattern p = Pattern.compile("^\\s*<\\?.*\\?>");
305                            Matcher m = p.matcher(xml);
306                            if (m.lookingAt())
307                                    is.setCharacterStream(new StringReader(m.replaceFirst(m.group()
308                                                    + " <fakeroot>")
309                                                    + " </fakeroot>"));
310                            else
311                                    is.setCharacterStream(new StringReader("<fakeroot> " + xml
312                                                    + "</fakeroot>"));
313                            db.parse(is);
314                    } catch (Exception ex) {
315                            throw new MobbedException("XML is invalid\n" + ex.getMessage());
316                    }
317            }

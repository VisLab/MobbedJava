001    package edu.utsa.mobbed;
002    
003    import java.sql.Connection;
004    import java.sql.PreparedStatement;
005    import java.sql.ResultSet;
006    import java.sql.SQLException;
007    import java.sql.Types;
008    import java.util.HashMap;
009    import java.util.UUID;
010    
011    /**
012     * Handler class for EVENT_TYPES table. Event types facilitate tagging and
013     * association of events across multiple datasets.
014     * 
015     * @author Arif Hossain, Jeremy Cockfield, Kay Robbins
016     * 
017     */
018    public class EventTypes {
019    
020            private Connection dbCon;
021            private UUID eventTypeUuid;
022            private String eventType;
023            private String eventTypeDescription;
024            private HashMap<String, UUID> etMap;
025    
026            /**
027             * Creates a EventTypes object.
028             * 
029             * @param dbCon
030             *            a connection to the database
031             */
032            public EventTypes(Connection dbCon) {
033                    this.dbCon = dbCon;
034                    this.eventTypeUuid = null;
035                    this.eventType = null;
036                    this.eventTypeDescription = null;
037                    etMap = new HashMap<String, UUID>();
038            }
039    
040            /**
041             * Adds a event type to the hashmap.
042             * 
043             * @param eventType
044             *            the name of the event type
045             * @param eventTypeUuid
046             *            the UUID of the event type
047             */
048            public void addEventType(String eventType, UUID eventTypeUuid) {
049                    etMap.put(eventType.toUpperCase(), eventTypeUuid);
050            }
051    
052            /**
053             * Checks if the hashmap contains the event type.
054             * 
055             * @param eventType
056             *            the name of the event type
057             * @return true if the hashmap contains the event type, false if otherwise
058             */
059            public boolean containsEventType(String eventType) {
060                    return etMap.containsKey(eventType);
061            }
062    
063            /**
064             * Gets the event type UUID from the hashmap.
065             * 
066             * @param eventType
067             *            the name of the event type
068             * @return the UUID of the event type
069             */
070            public UUID getEventTypeUuid(String eventType) {
071                    return etMap.get(eventType);
072            }
073    
074            /**
075             * Gets the event type UUID from the event type.
076             * 
077             * @return the UUID of the event type
078             */
079            public UUID getEventTypeUuid() {
080                    return eventTypeUuid;
081            }
082    
083            /**
084             * Gets the string representation of the event type UUIDs.
085             * 
086             * @return the string representation of the event type UUIDs
087             */
088            public String[] getStringValues() {
089                    int mapSize = etMap.size();
090                    String[] eventTypeUuids = new String[mapSize];
091                    Object[] mapObject = etMap.values().toArray();
092                    for (int i = 0; i < mapSize; i++)
093                            eventTypeUuids[i] = mapObject[i].toString();
094                    return eventTypeUuids;
095            }
096    
097            /**
098             * Sets the class fields of a Event Type object.
099             * 
100             * @param eventType
101             *            the name of the event type
102             * @param eventTypeDescription
103             *            the description of the event type
104             */
105            public void reset(String eventType, String eventTypeDescription) {
106                    this.eventTypeUuid = UUID.randomUUID();
107                    this.eventType = eventType;
108                    this.eventTypeDescription = eventTypeDescription;
109            }
110    
111            /**
112             * Retrieves the event types from the database and puts them into a hashmap.
113             * 
114             * @param eventTypeUuids
115             *            the UUIDs of the event types that exist in the database
116             * @throws MobbedException
117             *             if an error occurs
118             */
119            public void retrieveMap(String[] eventTypeUuids) throws MobbedException {
120                    String query = "SELECT EVENT_TYPE, EVENT_TYPE_UUID FROM EVENT_TYPES WHERE EVENT_TYPE_UUID IN (";
121                    int numUuids = eventTypeUuids.length;
122                    for (int i = 0; i < numUuids - 1; i++)
123                            query += "?,";
124                    query += "?)";
125                    try {
126                            PreparedStatement pStmt = dbCon.prepareStatement(query);
127                            for (int i = 0; i < numUuids; i++)
128                                    pStmt.setObject(i + 1, UUID.fromString(eventTypeUuids[i]),
129                                                    Types.OTHER);
130                            ResultSet rs = pStmt.executeQuery();
131                            while (rs.next())
132                                    etMap.put(rs.getString(1).toUpperCase(),
133                                                    UUID.fromString(rs.getString(2)));
134                    } catch (SQLException ex) {
135                            throw new MobbedException(
136                                            "Could not retrieve the event types to put in the hashmap\n"
137                                                            + ex.getMessage());
138                    }
139            }
140    
141            /**
142             * Saves all event types as a batch. All event types in the batch will be
143             * successfully written or the operation will be aborted.
144             * 
145             * @throws MobbedException
146             *             if an error occurs
147             */
148            public void save() throws MobbedException {
149                    String insertQry = "INSERT INTO EVENT_TYPES "
150                                    + "(EVENT_TYPE_UUID, EVENT_TYPE, EVENT_TYPE_DESCRIPTION)"
151                                    + " VALUES (?,?,?)";
152                    try {
153                            PreparedStatement pStmt = dbCon.prepareStatement(insertQry);
154                            pStmt.setObject(1, eventTypeUuid, Types.OTHER);
155                            pStmt.setString(2, eventType);
156                            pStmt.setString(3, eventTypeDescription);
157                            pStmt.executeUpdate();
158                    } catch (SQLException ex) {
159                            throw new MobbedException("Could not save the event types\n"
160                                            + ex.getMessage());
161                    }
162            }
163    
164    }
